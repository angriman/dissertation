\caption{\footnotesize Static iterative \suitor algorithm~\cite{DBLP:conf/ipps/ManneH14}}
\label{algo:iter-suitor}

\For{\textbf{each} $u \in V$}

\State$ws(u) \gets 0$
\EndFor
\For{\textbf{each} $u \in V$}
\State$\textsc{findSuitor}(u)$\Comment{\Cref{algo:iter-find-suitor}}
\EndFor
\end{algorithmic}
\end{algorithm}
